# 一、向量(vector) #

```bash
> A <- c(1,2,3,4,5,6)
> B <- c("A","B","C","D","E")
> A
[1] 1 2 3 4 5 6
> B
[1] "A" "B" "C" "D" "E"
> class(A)
[1] "numeric"
> mode(A)
[1] "numeric"
> typeof(A)
[1] "double"
> class(B)
[1] "character"
> mode(B)
[1] "character"
> typeof(B)
[1] "character"
```

# 二、矩阵（matrix）#

```bash
> var1=sample(31:60,25,replace=T)
# 矩阵的元素类型是一致的，或数值，或字符串；维度为二，即二维数组；
> C <- matrix(var1,nrow=5)
> C
[,1] [,2] [,3] [,4] [,5]
[1,]   43   32   54   41   56
[2,]   56   45   37   57   38
[3,]   50   50   40   32   50
[4,]   34   56   49   36   42
[5,]   56   41   33   40   46

> class(C)
[1] "matrix"
> mode(C)
[1] "numeric"
> typeof(C)
[1] "integer"
```

# 四、数据框（data.frame）#

```bash
> var3 <- LETTERS[1:8]
> var4 <- sample(3:12,8,replace=T)
# 数据框的元素类型可不一致，但组成数据框的向量的维数必须相等；
> D <- data.frame(var3,var4)
> D
var3 var4
1    A   11
2    B   10
3    C    9
4    D    6
5    E   10
6    F   11
7    G    7
8    H   10
# 查看类型；
> class(D)
[1] "data.frame"
> mode(D)
[1] "list"
> typeof(D)
[1] "list"
```
